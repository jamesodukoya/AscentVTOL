# X500 Quadcopter - High-rate attitude and position
# These commands are executed in PX4 console after startup (refresh rate - Hz)

mavlink stream -d /dev/ttyS0 -s ATTITUDE -r 100
mavlink stream -d /dev/ttyS0 -s ATTITUDE_QUATERNION -r 100
mavlink stream -d /dev/ttyS0 -s LOCAL_POSITION_NED -r 50
mavlink stream -d /dev/ttyS0 -s GLOBAL_POSITION_INT -r 10
mavlink stream -d /dev/ttyS0 -s HIGHRES_IMU -r 50
mavlink stream -d /dev/ttyS0 -s GPS_RAW_INT -r 10
mavlink stream -d /dev/ttyS0 -s BATTERY_STATUS -r 5
mavlink stream -d /dev/ttyS0 -s SYS_STATUS -r 2
mavlink stream -d /dev/ttyS0 -s HEARTBEAT -r 1
